Why RSpec?

Clear and readable

          describe Stack do
            subject { Stack.new }

            context "on creation" do
              it { is_empty }
            end
          end

vs

          class StackTest < Test::Unit
            def setup
              @stack = Stack.new
            end

            def test_is_empty_on_creation
              assert_equal true, @stack.empty?
            end
          end
