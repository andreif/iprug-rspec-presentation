Why RSpec?

Clear, concise and readable

            class StackTest < Test::Unit
              def setup
                @stack = Stack.new
              end

              def test_new_is_empty
                assert_equal true, @stack.empty?
              end

              def test_new_has_size_0
                assert_equal 0, @stack.size
              end
            end

vs

 
            describe Stack do
              context "when new" do
                subject { Stack.new }

                it { is_empty }
                it { has(0).items }
              end
            end
